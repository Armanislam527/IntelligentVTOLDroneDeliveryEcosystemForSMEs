% Section 1: Introduction
\section{Introduction}

\begin{frame}{The Challenge for Bangladesh's SMEs}
\begin{block}{Critical Industry Challenges}
\item \textbf{Inefficient Logistics:} 89\% of SMEs struggle with costly last-mile delivery
\item \textbf{Unreliable in Bad Weather:} 91\% report delayed deliveries during monsoon season
\item \textbf{High Operational Costs:} Delivery costs consume 18-25\% of operational budgets
\item \textbf{Limited Business Intelligence:} 73\% lack real-time analytics for inventory and forecasting
\item \textbf{Environmental Impact:} Traditional delivery systems contribute to carbon emissions
\end{block}
\end{frame}

% Section 2: Vision & Mission
\section{Vision \& Mission}

\begin{frame}{Vision \& Mission}
\begin{columns}[T]
\column{0.5\textwidth}
\begin{block}{\textcolor{greenlight}{Vision}}
To build a scalable, green logistics network that transforms last-mile delivery and business intelligence for millions of SMEs across South Asia.

\begin{block}{\textcolor{white}{Mission}}

% Section 3: Solution
\section{Solution}

\begin{frame}{GreenLift AI: The Integrated Solution}
\begin{block}{A Revolutionary Approach}
GreenLift AI is an integrated, eco-friendly drone delivery and AI-powered business intelligence platform designed specifically for SMEs in Bangladesh.
\end{block}

\begin{columns}[T]
\column{0.55\textwidth}
\item \textbf{Solar-Powered VTOL Drones:} Vertical take-off and landing with solar energy harvesting
\item \textbf{AI Dashboard:} Real-time business insights for inventory, sales, and customer analytics
\item \textbf{Zero-Carbon Logistics:} Eliminates carbon footprint while reducing delivery costs
\item \textbf{Complete Solution:} One-time setup cost without recurring wages or salary
\item \textbf{Smart Tension System:} Automatic package release if weight exceeds safe limits
